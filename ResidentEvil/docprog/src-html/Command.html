003 */
004public class Command {
005    private String commandWord;
006    private String secondWord;
007    
008    /**
009     * Create a command object. First and second word must be supplied, but
010     * either one (or both) can be null.
011     * @param firstWord The first word of the command. Null if the command
012     *                  was not recognised.
013     * @param secondWord The second word of the command.
014     */
015    public Command(String firstWord, String secondWord)
016    {
017        commandWord = firstWord;
018        this.secondWord = secondWord;
019    }
020
021    /**
022     * Return the command word (the first word) of this command. If the
023     * command was not understood, the result is null.
024     * @return The command word.
025     */
026    public String getCommandWord()
027    {
028        return commandWord;
029    }
030
031    /**
032     * @return The second word of this command. Returns null if there was no
033     * second word.
034     */
035    public String getSecondWord()
036    {
037        return secondWord;
038    }
039
040    /**
041     * @return true if this command was not understood.
042     */
043    public boolean isUnknown()
044    {
045        return (commandWord == null);
046    }
047
048    /**
049     * @return true if the command has a second word.
050     */
051    public boolean hasSecondWord()
052    {
053        return (secondWord != null);
054    }
055}
