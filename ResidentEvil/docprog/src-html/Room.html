001import java.util.*;
002
003/**
004 * Room est une classe permettant d'instancier un object "Room" (piece)
005 * avec des sorties.
006 */
007
008public class Room {
009    private String description;
010    private String imageName;
011    private ArrayList<Zombie> zombies;
012    private HashMap<String, Room> exits;
013    private ArrayList<Item> items;
014    private HashMap<String, Boolean> states;
015    private HashMap<String, Item> itemsNeeded;
016
017    /**
018     * Crée une pièce avec une description.
019     * @param description The room's description.
020     */
021    public Room(String description, String image) {
022        this.description = description;
023        zombies = new ArrayList<Zombie>();
024        exits = new HashMap<String, Room>();
025        items = new ArrayList<Item>();
026        states = new HashMap<String, Boolean>();
027        itemsNeeded = new HashMap<String, Item>();
028        imageName = image;
029    }
030    
031    public ArrayList<Zombie> getZombies() {
032        return zombies;
033    }
034    
035    public void setZombies(ArrayList<Zombie> zombies) {
036        this.zombies = zombies;
037    }
038    
039    public void setZombiesInt(int nb){
040        zombies.clear();
041        for (int i=0; i<nb; i++) {
042                Zombie zombie = new Zombie(2);          
043                zombies.add(zombie);
044        }
045    }
046    
047    public HashMap<String, Room> getExits() {
048        return exits;
049    }
050    
051    public void setExits(HashMap<String, Room> exits) {
052        this.exits = exits;
053    }
054
055    public ArrayList<Item> getItems() {
056                return items;
057        }
058
059        public void setItems(ArrayList<Item> items) {
060                this.items = items;
061        }
062        
063        public void addItem(Item item) {
064                items.add(item);
065        }
066        
067        public void removeItem(Item item) {
068                items.remove(item);
069        }
070        
071        public boolean haveItemName(String name) {
072                for(int i=0;i<items.size();i++) {
073                        if(items.get(i).getName().equals(name))
074                                return true;
075                }
076                return false;
077        }
078
079        public void setDescription(String description) {
080        this.description = description;
081    }
082        
083        public HashMap<String, Boolean> getStates() {
084                return states;
085        }
086        
087        public void setStates(HashMap<String, Boolean> states){
088                this.states = states;
089        }
090        
091        public boolean getState(String direction) {
092                return states.get(direction);
093        }
094        
095        public void setState(String direction, Boolean state){
096                this.states.put(direction, state);
097        }
098
099    public HashMap<String, Item> getItemsNeeded() {
100                return itemsNeeded;
101        }
102
103        public void setItemsNeeded(HashMap<String, Item> itemsNeeded) {
104                this.itemsNeeded = itemsNeeded;
105        }
106        
107        public Item getItemNeeded(String direction) {
108                return itemsNeeded.get(direction);
109        }
110        
111        public void setItemNeeded(String direction, Item item){
112                this.itemsNeeded.put(direction, item);
113        }
114
115        /**
116     * getExit permet de retourner un objet room
117     * en fonction d'une direction.
118     * @param {String} direction
119     * @return {Room} - retourne la sortie dans la direction indiqué
120     */
121    public Room getExit(String direction){
122        return exits.get(direction);
123    }
124
125    /**
126     * getExitString permet de retourner en chaine de caractère
127     * l'ensemble des sorties de la piece courante.
128     * @return {String} - indique l'ensemble des sorties.
129     */
130    public String getExitString(){
131        StringBuilder exit = new StringBuilder("Exits: ");
132        //exit.append("Exits: ");
133        for (String key : exits.keySet())
134                exit.append(" " + key);
135        return exit.toString();
136    }
137
138    /**
139     * Permet de fixer une sortie.
140     * @method setExit
141     * @param {String} direction - clé
142     * @param {Room} neighbor - valeur
143     */
144    public void setExit(String direction, Room neighbor) {
145        exits.put(direction, neighbor);
146    }
147    
148    // return the opposite direction from where the player is going
149    public String getOpposite(String direction) {
150        String opposite ="";
151        if(direction.matches("north"))
152                opposite="south";
153        if(direction.matches("south"))
154                opposite="north";
155        if(direction.matches("east"))
156                opposite="west";
157        if(direction.matches("west"))
158                opposite="east";
159        if(direction.matches("up"))
160                opposite="down";
161        if(direction.matches("down"))
162                opposite="up";
163        return opposite;
164    }
165
166    /**
167     * Permet de récupérer la description courante.
168     * @method getDescription
169     * @return {String} - description de l'objet courant
170     */
171    public String getDescription(){
172        return description;
173    }
174
175
176        /**
177     * Permet de récupérer la description courante,
178     * ainsi que la liste des sorties. 
179     * @method getLongDescription
180     * @return {String} - description de l'objet courant et les sorties.
181     */   
182    public String getLongDescription(){
183        StringBuilder longDescription = new StringBuilder("You are " + getDescription() + "\n");
184        if(zombies.size() > 0) {
185                longDescription.append(zombies.size() + " yellow jackets" + "\n");
186        }
187        if(!items.isEmpty()) {
188                longDescription.append("Some items are here !" + "\n");
189                for(int i=0;i<items.size();i++) {
190                        longDescription.append(" *" + items.get(i).getName() +"* \n");
191                }
192        }
193        longDescription.append(getExitString());
194        return longDescription.toString();
195    }
196    
197    public String getImageName()
198        {
199                return imageName;
200        }
201}
