001import javax.swing.*;
002import java.awt.*;
003import java.awt.event.*;
004import java.net.URL;
005import java.awt.image.*;
006
007/**
008 * This class implements a simple graphical user interface with a text entry
009 * area, a text output area and an optional image.
010 * 
011 * @author Michael Kolling
012 * @version 1.0 (Jan 2003)
013 */
014public class UserInterface implements ActionListener
015{
016    private GameEngine engine;
017    private JFrame myFrame;
018    private JTextField entryField;
019    private JTextArea log;
020    private JLabel image;
021
022    /**
023     * Construct a UserInterface. As a parameter, a Game Engine
024     * (an object processing and executing the game commands) is
025     * needed.
026     * 
027     * @param gameEngine  The GameEngine object implementing the game logic.
028     */
029    public UserInterface(GameEngine gameEngine)
030    {
031        engine = gameEngine;
032        createGUI();
033    }
034
035    /**
036     * Print out some text into the text area.
037     */
038    public void print(String text)
039    {
040        log.append(text);
041        log.setCaretPosition(log.getDocument().getLength());
042    }
043
044    /**
045     * Print out some text into the text area, followed by a line break.
046     */
047    public void println(String text)
048    {
049        log.append(text + "\n");
050        log.setCaretPosition(log.getDocument().getLength());
051    }
052
053    /**
054     * Show an image file in the interface.
055     */
056    public void showImage(String imageName)
057    {
058        URL imageURL = this.getClass().getClassLoader().getResource(imageName);
059        if(imageURL == null)
060            System.out.println("image not found");
061        else {
062            ImageIcon icon = new ImageIcon(imageURL);
063            image.setIcon(icon);
064            myFrame.pack();
065        }
066    }
067
068    /**
069     * Enable or disable input in the input field.
070     */
071    public void enable(boolean on)
072    {
073        entryField.setEditable(on);
074        if(!on)
075            entryField.getCaret().setBlinkRate(0);
076    }
077
078    /**
079     * Set up graphical user interface.
080     */
081    private void createGUI()
082    {
083        myFrame = new JFrame("Residu Evil");
084        entryField = new JTextField(34);
085
086        log = new JTextArea();
087        log.setEditable(false);
088        JScrollPane listScroller = new JScrollPane(log);
089        listScroller.setPreferredSize(new Dimension(800, 600));
090        listScroller.setMinimumSize(new Dimension(100,100));
091
092        JPanel panel = new JPanel();
093        image = new JLabel();
094
095        panel.setLayout(new BorderLayout());
096        panel.add(image, BorderLayout.NORTH);
097        panel.add(listScroller, BorderLayout.CENTER);
098        panel.add(entryField, BorderLayout.SOUTH);
099
100        myFrame.getContentPane().add(panel, BorderLayout.CENTER);
101
102        // add some event listeners to some components
103        myFrame.addWindowListener(new WindowAdapter() {
104            public void windowClosing(WindowEvent e) {System.exit(0);}
105        });
106
107        entryField.addActionListener(this);
108
109        myFrame.pack();
110        myFrame.setVisible(true);
111        entryField.requestFocus();
112    }
113
114    /**
115     * Actionlistener interface for entry textfield.
116     */
117    public void actionPerformed(ActionEvent e) 
118    {
119        // no need to check the type of action at the moment.
120        // there is only one possible action: text entry
121        processCommand();
122    }
123
124    /**
125     * A command has been entered. Read the command and do whatever is 
126     * necessary to process it.
127     */
128    private void processCommand()
129    {
130        boolean finished = false;
131        String input = entryField.getText();
132        entryField.setText("");
133
134        engine.interpretCommand(input);
135    }
136}
