001import java.util.ArrayList;
002
003public class Item {
004
005        private String name;
006        private String description;
007        private int number;
008        public int initial_number;
009
010        public Item(String name, String description, int number) {
011                this.name = name;
012                this.description = description;
013                this.number = number;
014                initial_number = number;
015        }
016
017        public String getName() {
018                return name;
019        }
020
021        public void setName(String name) {
022                this.name = name;
023        }
024
025        public int getNumber() {
026                return number;
027        }
028
029        public void setNumber(int number) {
030                this.number = number;
031        }
032        
033        public String getDescription() {
034                return description;
035        }
036
037        public void setDescription(String description) {
038                this.description = description;
039        }
040
041        static public Item findByName(String s, ArrayList<Item> L) {
042        int n=0;
043        while (n < L.size()) {
044            Item i = L.get(n);
045            if (s.equals(i.getName())) 
046                return i;
047            n++;
048        }
049        return null;
050    }
051        
052}
