001/**
002 * CommandWords permet de verifié la véracité des commandes.
003 */
004public class CommandWords {
005    // a constant array that holds all valid command words
006    private static final String[] validCommands = {
007        "go", "quit", "help", "shoot", "look", "pickup", "use", "player", "unlock", "reload", "info", "back"
008    };
009    
010    /**
011     * Constructeur - initialise les commandes.
012     */
013    public CommandWords() {
014        // nothing to do at the moment...
015    }
016
017    /**
018     * Regarde si une chaine de caractère est une commande valide.
019     * @method isCommand
020     * @param {String} aString -
021     * @return {boolean} - true si la chaine de caractère en
022     *  parametre est une commande valide, false sinon
023     */
024    public boolean isCommand(String aString) {
025        for(int i = 0; i < validCommands.length; i++) {
026            if(validCommands[i].equals(aString)) return true;
027        }
028        return false;
029    }
030
031    /**
032     * Permet d'afficher la liste de toutes les commandes
033     * reconnu par le jeu.
034     * @method showAll
035     */
036    public String getCommandList() 
037    {
038        StringBuilder commands = new StringBuilder();
039        for(int i = 0; i < validCommands.length; i++) {
040            commands.append( validCommands[i] + "  " );
041        }
042        return commands.toString();
043    }
044}
