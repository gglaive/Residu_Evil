001import java.awt.Graphics;
002import javax.swing.JPanel;
003import java.awt.Image;
004import java.io.File;
005import java.io.IOException;
006import javax.imageio.ImageIO;
007
008public class Panneau extends JPanel { 
009  public void paintComponent(Graphics g){
010          try {
011                  Image img = ImageIO.read(new File("plan.png"));
012                  g.drawImage(img, 0, 0, this.getWidth(), this.getHeight(), this); 
013          } catch (IOException e) {
014              e.printStackTrace();
015          } 
016  }               
017}
