001import java.util.Random;
002
003public class Zombie extends Character{
004
005        int aggro;
006        
007    public Zombie(int health){
008        super(health);
009        this.aggro = 0;
010    }
011    
012        public int getAggro() {
013                return aggro;
014        }
015
016
017        public void setAggro(int aggro) {
018                this.aggro = aggro;
019        }
020
021
022        public String attack(Player player){
023                Random rand = new Random();
024                int random = rand.nextInt(10);
025                StringBuilder string = new StringBuilder("");
026                if((this.aggro >= 100) || (random == 9)){
027                        player.setHealth(player.getHealth() -1);
028                        string.append("The zombie attacks ! \n");
029                this.aggro = 0;
030                }
031                
032        else {
033                switch(random) {
034                        
035                case 1:
036                        string.append("The zombie is just looking upward \n");
037                        this.aggro += 10;
038                        break;
039                
040                case 2:
041                        string.append("The zombie stands still \n");
042                        this.aggro += 15;
043                        break;
044                        
045                case 3:
046                        string.append("The zombie is looking at you \n");
047                        this.aggro += 25;
048                        break;
049                        
050                case 4:
051                        string.append("The zombie is moving slowly \n");
052                        this.aggro += 30;
053                        break;
054                        
055                case 5:
056                        string.append("The zombie screams loudly \n");
057                        this.aggro += 40;
058                        break;
059                        
060                case 6:
061                        string.append("The zombie raises its arm and tries to reach you \n");
062                        this.aggro += 50;
063                        break;
064                        
065                case 7:
066                        string.append("The zombie is heading towards you! \n");
067                        this.aggro += 85;
068                        break;
069                        
070                case 8:
071                        string.append("The zombie is running at you! \n");
072                        this.aggro += 100;
073                        break;
074
075                default:
076                        string.append("The zombie is not paying attention \n");
077                        break;          
078                }       
079                string.append("Aggro: " + this.aggro + "% \n"); 
080        }
081                return string.toString();
082        }
083
084}
